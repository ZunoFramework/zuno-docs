#include "routes/users.hpp"

void registerUserRoutes(zuno::App& app) {
  app.get("/users", [](Request& req, Response& res) {
    res.json({
      { "id", 1 },
      { "name", "Alice" }
    });
  });

  app.get("/users/:id", [](auto& req, auto& res) {
    std::string id = req.params["id"];
    res.json({
      { "id", id },
      { "name", "Usuario " + id }
    });
  });
}`}
