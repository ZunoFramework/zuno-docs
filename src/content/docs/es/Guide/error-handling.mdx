##  `main.cpp`

<Code
  lang="cpp"
  code={`#include <zuno/zuno.hpp>
#include <stdexcept>

int main() {
  zuno::App app;

  // Middleware de manejo de errores
  app.use([](auto& req, auto& res, zuno::Next next) {
    try {
      next();
    } catch (const std::exception& ex) {
      res.status(500).json({
        { "error", "Error interno del servidor" },
        { "detalle", ex.what() }
      });
    }
  });
