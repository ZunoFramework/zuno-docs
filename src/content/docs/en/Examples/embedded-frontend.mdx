## ğŸ“„ `main.cpp`

<Code
  lang="cpp"
  code={`#include <zuno/zuno.hpp>
#include <fstream>
#include <sstream>

int main() {
  zuno::App app;

  // Serve static frontend files (JS, CSS, images)
  app.use(staticFiles("public"));

  // Fallback route for SPAs
  app.get("/*", [](Request& req, Response& res) {
    res.html(zuno::readFile("./public/index.html"));
  });

  app.listen(3000);
}`}
/>
